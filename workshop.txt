description=[h1]Features[/h1]
description=[list]
description=[*]Prevent destroying
description=[*]Prevent disassembling
description=[*]Prevent picking-up containers (see exceptions below)
description=[*]Sandbox option to configure who can destroy objects
description=[*]Sandbox option to configure who can destroy objects inside a safehouse
description=[/list]
description=
description=Picking-up is still allowed for the following containers:
description=[list]
description=[*]Barbecues
description=[*]Bins / Trash Cans
description=[*]Cash Registers
description=[*]Fireplaces
description=[*]Freezers
description=[*]Fridges
description=[*]Microwaves
description=[*]Stoves
description=[*]Woodstoves
description=[/list]
